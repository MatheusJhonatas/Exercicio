47) Considere um vetor de 10 números inteiros positivos maiores que zero e um único número inteiro, também positivo
e maior que zero. 
Faça um programa para:
a. ler pelo teclado o vetor;
b. ler pelo teclado o número X;
c. dizer quantos números no vetor são maiores que X, menores que X e iguais a X.

using System;

class Program
{
    static void Main(string[] args)
    {
        int[] vetor = new int[10];
        int x;

        // Ler o vetor
        Console.WriteLine("Digite 10 números inteiros positivos maiores que zero:");
        for (int i = 0; i < vetor.Length; i++)
        {
            vetor[i] = LerInteiroPositivo();
        }

        // Ler o número X
        Console.WriteLine("Digite um número inteiro positivo maior que zero (X):");
        x = LerInteiroPositivo();

        // Contagem dos números no vetor
        int maioresQueX = 0;
        int menoresQueX = 0;
        int iguaisAX = 0;

        foreach (int num in vetor)
        {
            if (num > x)
                maioresQueX++;
            else if (num < x)
                menoresQueX++;
            else
                iguaisAX++;
        }

        // Exibir os resultados
        Console.WriteLine($"No vetor, há {maioresQueX} número(s) maior(es) que X, {menoresQueX} número(s) menor(es) que X e {iguaisAX} número(s) igual(is) a X.");
    }

    static int LerInteiroPositivo()
    {
        int num;
        while (true)
        {
            if (int.TryParse(Console.ReadLine(), out num) && num > 0)
            {
                return num;
            }
            else
            {
                Console.WriteLine("Por favor, digite um número inteiro positivo maior que zero:");
            }
        }
    }
}
