36) Faça um programa que leia 10 valores inteiros e positivos e:
- Encontre o maior valor
- Encontre o menor valor
- Calcule a média dos números lidos

using System;

class Program
{
    static void Main(string[] args)
    {
        const int quantidadeNumeros = 10;
        int[] numeros = new int[quantidadeNumeros];
        int maior = int.MinValue;
        int menor = int.MaxValue;
        int soma = 0;

        // Solicita que o usuário insira os valores
        for (int i = 0; i < quantidadeNumeros; i++)
        {
            Console.Write($"Insira o {i + 1}º valor: ");
            numeros[i] = Convert.ToInt32(Console.ReadLine());

            // Verifica se o número inserido é positivo
            if (numeros[i] < 0)
            {
                Console.WriteLine("O número deve ser positivo. Tente novamente.");
                i--; // Volta uma iteração para permitir que o usuário insira o valor corretamente
            }
            else
            {
                // Atualiza o maior e o menor número
                maior = Math.Max(maior, numeros[i]);
                menor = Math.Min(menor, numeros[i]);

                // Soma os valores para calcular a média
                soma += numeros[i];
            }
        }

        // Calcula a média
        double media = (double)soma / quantidadeNumeros;

        // Exibe os resultados
        Console.WriteLine($"O maior valor é: {maior}");
        Console.WriteLine($"O menor valor é: {menor}");
        Console.WriteLine($"A média dos números é: {media}");
    }
}
