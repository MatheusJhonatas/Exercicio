28) Faça um programa que gera e escreve os números ímpares dos números lidos entre 100 e 200.

Meu código:
using System;

namespace Exercicio28
{
    public class Exercicio28
    {
        public static void Main()
        {
            //28) Faça um programa que gera e escreve os números ímpares dos números lidos entre 100 e 200.
            Console.WriteLine("Os números impares entre 100 e 200 são: ");
            for (var i = 100; i <= 200; i++)
            {

                while (i >= 100 && i <= 200 && i % 2 != 0)
                {

                    Console.WriteLine(i);
                    i++;
                }

            }
        }
    }
}

código do chatgpt:

using System;

class Program
{
    static void Main()
    {
        Console.Write("Digite a quantidade de números a serem processados: ");
        int quantidadeNumeros = int.Parse(Console.ReadLine());

        for (int i = 1; i <= quantidadeNumeros; i++)
        {
            Console.Write($"Digite o {i}º número entre 100 e 200: ");
            int numero = int.Parse(Console.ReadLine());

            // Verificar se o número está no intervalo de 100 a 200 e é ímpar
            if (numero >= 100 && numero <= 200 && numero % 2 != 0)
            {
                // Exibir o número ímpar
                Console.WriteLine($"O número {numero} é ímpar\n");
            }
            else
            {
                Console.WriteLine($"O número {numero} não está no intervalo de 100 a 200 ou não é ímpar.\n");
            }
        }

        Console.ReadLine(); // Aguardar a entrada do usuário antes de fechar a janela
    }
}
